'''
const title = document.querySelector("#title");

const BASE = "black";

function hadnClick() {
  const current = title.style.color;
  if (current === "black") {
    title.style.color = "salmon";
  } else {
    title.style.color = "black";
  }
}

function init() {
  title.style.color = BASE;
  title.addEventListener("click", hadnClick);
}
init();
'''
